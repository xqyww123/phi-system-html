section ‹Semantics›

debt_axiomatization sem_tup_T    :: ‹TY list ⇒ TY›
                and sem_mk_tup   :: ‹VAL list ⇒ VAL›
                and sem_dest_tup :: ‹VAL ⇒ VAL list›
  where sem_dest_mk_tup[simp]: ‹sem_dest_tup (sem_mk_tup vs) = vs›
  and   WT_tup[simp]: ‹Well_Type (sem_tup_T ts)  = { sem_mk_tup vs |vs. list_all2 (λ t v. v ∈ Well_Type t) ts vs }›
  and   zero_tup[simp]: ‹Zero (sem_tup_T Ts)     = map_option sem_mk_tup (those (map Zero Ts))›
  and   V_tup_sep_disj_R[simp]: ‹sem_mk_tup l1 ## vl2 ⟷ (∃l2. vl2 = sem_mk_tup l2)›
  and   V_tup_sep_disj_L[simp]: ‹vl1 ## sem_mk_tup l2 ⟷ (∃l1. vl1 = sem_mk_tup l1)›
  and   V_tup_mult    : ‹sem_mk_tup l1 * sem_mk_tup l2 = sem_mk_tup (l1 @ l2)›
  and   idx_step_type_tup [eval_aggregate_path] : ‹i < length tys ⟹ idx_step_type (AgIdx_N i) (sem_tup_T tys) = tys!i ›
  and   valid_idx_step_tup[eval_aggregate_path] : ‹valid_idx_step (sem_tup_T tys) j ⟷ j ∈ {AgIdx_N i | i. i < length tys}›
  and   idx_step_value_tup[eval_aggregate_path] : ‹idx_step_value (AgIdx_N i) (sem_mk_tup vs) = vs!i›
  and   idx_step_mod_value_tup : ‹idx_step_mod_value (AgIdx_N i) f (sem_mk_tup vs) = sem_mk_tup (vs[i := f (vs!i)])›

lemma sem_mk_tup_inj[simp]:
  ‹sem_mk_tup vs1 = sem_mk_tup vs2 ⟷ vs1 = vs2›
  by (metis sem_dest_mk_tup)

  end)
