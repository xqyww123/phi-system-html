Theory Phi_Algebras.Phi_Algebra_Set

theory Phi_Algebra_Set
  imports Algebras
begin

subsection ‹Separation Closed Set›

definition One_in_Set :: ‹'a::one set ⇒ bool›
  where ‹One_in_Set S ⟷ 1 ∈ S›

definition Sep_Closed :: ‹'a::sep_magma set ⇒ bool›
  where ‹Sep_Closed S ⟷ (∀x y. x ## y ⟶ x ∈ S ∧ y ∈ S ⟶ (x * y) ∈ S)›

lemma Sep_Closed_inter:
  ‹Sep_Closed S1 ⟹ Sep_Closed S2 ⟹ Sep_Closed (S1 ∩ S2)›
  unfolding Sep_Closed_def by blast

lemma One_in_Set_inter:
  ‹One_in_Set S1 ⟹ One_in_Set S2 ⟹ One_in_Set (S1 ∩ S2)›
  unfolding One_in_Set_def by blast

lemma Sep_Closed_UNIV[simp]:
  ‹Sep_Closed UNIV›
  unfolding Sep_Closed_def by simp

lemma One_in_Set_UNIV[simp]:
  ‹One_in_Set UNIV›
  unfolding One_in_Set_def by simp

typedef (overloaded) ('a::sep_magma_1) sep_closed_set
    = ‹Collect (Sep_Closed::'a set ⇒ bool) ∩ Collect One_in_Set›
  morphisms dest_sep_closed_set sep_closed_set
  unfolding Sep_Closed_def One_in_Set_def by blast

setup_lifting type_definition_sep_closed_set

lift_definition sep_closed_member :: ‹'a::sep_magma_1 ⇒ 'a sep_closed_set ⇒ bool› (infix "∈⇩S" 50)
  is ‹λx S. x ∈ S› .

lemma in_sep_closed_set[simp]:
  ‹Sep_Closed S ⟹ One_in_Set S ⟹ x ∈⇩S sep_closed_set S ⟷ x ∈ S›
  unfolding sep_closed_member_def
  by (simp add: sep_closed_set_inverse)

lemma one_in_sep_closed_set[simp]:
  ‹1 ∈⇩S S› for S :: ‹'a::sep_magma_1 sep_closed_set›
  by (transfer; simp add: One_in_Set_def)

lemma mult_in_sep_closed_set[simp]:
  ‹x ## y ⟹ x ∈⇩S S ∧ y ∈⇩S S ⟶ x * y ∈⇩S S› for S :: ‹'a::sep_algebra sep_closed_set›
  by (transfer; simp add: Sep_Closed_def)

lift_definition sep_closed_inter :: ‹'a::sep_magma_1 sep_closed_set ⇒ 'a sep_closed_set ⇒ 'a sep_closed_set› (infixl "∩⇩S" 65)
  is ‹λS1 S2. S1 ∩ S2›
  by (clarsimp simp add: Sep_Closed_def One_in_Set_def; blast)

definition sep_closed_image :: ‹('a::sep_algebra ⇒ 'b::sep_algebra) ⇒ 'a sep_closed_set ⇒ 'b sep_closed_set› (infixr "`⇩S" 90)
  where ‹(f `⇩S S) = sep_closed_set (f ` dest_sep_closed_set S) ›

lemma in_image_sep_closed[simp]:

  by (smt (verit, del_insts) Homo_Sep_Closed_def  dest_sep_closed_set dest_sep_closed_set_inverse image_iff in_sep_closed_set mem_Collect_eq sep_closed_image_def)

subsubsection ‹Common Sep-Closed Sets›

lemma sep_closed_partial_map[simp]:
  ‹Sep_Closed {vars. finite (dom vars)}›
  unfolding Sep_Closed_def
  by (clarsimp simp add: dom_mult)

lemma sep_closed_partial_map1[simp]:
  ‹Sep_Closed {h::'a ⇒ 'b :: sep_no_inverse. finite (dom1 h)}› 
  unfolding Sep_Closed_def
  by (clarsimp simp add: dom1_mult)

lemma Sep_Closed_pointwise:
  ‹ (∀k. P k 1)
⟹ (∀k x y. x ## y ⟶ P k x ∧ P k y ⟶ P k (x * y))
⟹   Sep_Closed {f. ∀k. P k (f k) }›
  unfolding Sep_Closed_def
  by (simp add: times_fun; blast)


subsection ‹Separation Homo Set›

definition Sep_Homo :: ‹'a::sep_magma_1 set ⇒ bool›
  where ‹Sep_Homo S ⟷ (∀x y. x ## y ⟶ (x * y) ∈ S ⟷ x ∈ S ∧ y ∈ S) ∧ 1 ∈ S›

lemma Sep_Homo_inter:
  ‹Sep_Homo S1 ⟹ Sep_Homo S2 ⟹ Sep_Homo (S1 ∩ S2)›
  unfolding Sep_Homo_def by blast

lemma Sep_Homo_UNIV[simp]:
  ‹Sep_Homo UNIV›
  unfolding Sep_Homo_def by simp

typedef (overloaded) ('a::sep_magma_1) sep_homo_set = ‹Collect (Sep_Homo::'a set ⇒ bool)›
  morphisms dest_sep_homo_set sep_homo_set
  unfolding Sep_Homo_def by blast

setup_lifting type_definition_sep_homo_set

lift_definition sep_homo_member :: ‹'a::sep_magma_1 ⇒ 'a sep_homo_set ⇒ bool› (infix "∈⇩S⇩H" 50)
  is ‹λx S. x ∈ S› .

lemma in_sep_homo_set[simp]:
  ‹Sep_Homo S ⟹ x ∈⇩S⇩H sep_homo_set S ⟷ x ∈ S›
  unfolding sep_homo_member_def
  by (simp add: sep_homo_set_inverse)

lemma one_in_sep_homo_set[simp]:
  ‹1 ∈⇩S⇩H S› for S :: ‹'a::sep_magma_1 sep_homo_set›
  by (transfer; simp add: Sep_Homo_def)

lemma mult_in_sep_homo_set[simp]:
  ‹x ## y ⟹ x * y ∈⇩S⇩H S ⟷ x ∈⇩S⇩H S ∧ y ∈⇩S⇩H S› for S :: ‹'a::sep_algebra sep_homo_set›
  by (transfer; simp add: Sep_Homo_def)

lift_definition sep_homo_inter :: ‹'a::sep_magma_1 sep_homo_set ⇒ 'a sep_homo_set ⇒ 'a sep_homo_set› (infixl "∩⇩S⇩H" 65)
  is ‹λS1 S2. S1 ∩ S2›
  by (clarsimp simp add: Sep_Homo_def; blast)

definition sep_homo_image :: ‹('a::sep_algebra ⇒ 'b::sep_algebra) ⇒ 'a sep_homo_set ⇒ 'b sep_homo_set› (infixr "`⇩S⇩H" 90)
  where ‹(f `⇩S⇩H S) = sep_homo_set (f ` dest_sep_homo_set S) ›

definition Homo_Sep_Homo :: ‹('a::sep_algebra ⇒ 'b::sep_algebra) ⇒ bool›
  where ‹Homo_Sep_Homo f ⟷ (∀S. Sep_Homo S ⟶ Sep_Homo (f ` S))›

lemma in_image_sep_homo[simp]:
  ‹Homo_Sep_Homo f ⟹ x ∈⇩S⇩H f `⇩S⇩H S ⟷ (∃x'. x = f x' ∧ x' ∈⇩S⇩H S)›
  by (smt (verit, best) Homo_Sep_Homo_def dest_sep_homo_set dest_sep_homo_set_inverse image_iff in_sep_homo_set mem_Collect_eq sep_homo_image_def)
  
subsubsection ‹Common Sep-Homo Sets›

lemma sep_homo_partial_map[simp]:
  ‹Sep_Homo {vars. finite (dom vars)}›
  unfolding Sep_Homo_def
  by (clarsimp simp add: dom_mult)

lemma sep_homo_partial_map1[simp]:
  ‹Sep_Homo {h::'a ⇒ 'b :: sep_no_inverse. finite (dom1 h)}› 
  unfolding Sep_Homo_def
  by (clarsimp simp add: dom1_mult)

lemma Sep_Homo_pointwise:
  ‹ (∀k. P k 1)
⟹ (∀k x y. x ## y ⟶ P k (x * y) ⟷ P k x ∧ P k y)
⟹   Sep_Homo {f. ∀k. P k (f k) }›
  unfolding Sep_Homo_def
  by (simp add: times_fun; blast)



end
