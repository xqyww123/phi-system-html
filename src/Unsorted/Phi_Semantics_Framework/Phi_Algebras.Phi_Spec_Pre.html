Theory Phi_Algebras.Phi_Spec_Pre

theory Phi_Spec_Pre
  imports Main
begin

definition Subjection :: " 'p set ⇒ bool ⇒ 'p set " (infixl "𝗌𝗎𝖻𝗃" 15)
  where " (T 𝗌𝗎𝖻𝗃 P) = {p. p ∈ T ∧ P}"

lemma Subjection_expn_set:
  "p ∈ (T 𝗌𝗎𝖻𝗃 P) ⟷ p ∈ T ∧ P"
  unfolding Subjection_def by simp

lemma Subjection_Id_on:
  ‹Id_on (S 𝗌𝗎𝖻𝗃 P) = (Id_on S 𝗌𝗎𝖻𝗃 P)›
  by (auto simp add: Subjection_expn_set)

lemma Subjection_image:
  ‹f ` (S 𝗌𝗎𝖻𝗃 P) = (f ` S 𝗌𝗎𝖻𝗃 P)›
  by (auto simp add: Subjection_expn_set)

definition ExSet :: " ('x ⇒ 'c set) ⇒ 'c set" (binder "∃*" 14)
  where "ExSet S = {p. (∃c. p ∈ S c)}"
notation ExSet (binder "∃⇧s" 14)

lemma ExSet_expn_set: "p ∈ (∃*c. S c) ⟷ (∃c. p ∈ S c)" unfolding ExSet_def by simp

lemma ExSet_Id_on:
  ‹Id_on (∃*x. S x) = (∃*x. Id_on (S x))›
  by (auto simp add: ExSet_expn_set; blast)

lemma ExSet_image:
  ‹f ` (∃*c. S c) = (∃*c. f ` S c)›
  by (auto simp add: ExSet_expn_set image_iff Bex_def; blast)
