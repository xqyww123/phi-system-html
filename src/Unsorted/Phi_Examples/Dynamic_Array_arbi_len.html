Theory Dynamic_Array_arbi_len

theory Dynamic_Array_arbi_len
  imports Phi_Semantics.PhiSem_C
          Phi_Semantics.PhiSem_Mem_C_AI
          PhiStd.PhiStd_Slice_a
          Phi_Semantics.PhiSem_Int_ArbiPrec
begin


φtype_def DynArr :: ‹address ⇒ TY ⇒ (VAL, 'x) φ ⇒ (fiction, 'x list) φ›
  where ‹l ⦂ DynArr addr TY T ≡ (a⇩D, len, cap) ⦂ 𝗆𝖾𝗆[addr] ⦃ data: ℙ𝗍𝗋 𝖺𝗋𝗋𝖺𝗒[cap] TY, len: ℕ, cap: ℕ ⦄❟
                                data ⦂ 𝗆𝖾𝗆[a⇩D] 𝔸𝗋𝗋𝖺𝗒[cap] T
                                𝗌𝗎𝖻𝗃 a⇩D len cap data. len = length l ∧ cap = length data ∧
                                                     len ≤ cap ∧ (cap = 0 ∨ cap < 2 * len) ∧
                                                     take len data = l ∧ address_to_base a⇩D ∧ address_to_base addr›

  deriving ‹Abstract_Domain T P ⟹ Abstract_Domain (DynArr addr TY T) (λl. list_all P l ∧ addr ≠ 0)›
       and ‹Object_Equiv T eq ⟹ Object_Equiv (DynArr addr TY T) (list_all2 eq)›
            (tactic: auto, subgoal' for x xa xb xc ‹rule exI[where x=‹xa @ drop (length xa) xc›]›)
       and ‹ 𝖼𝗈𝗇𝖽𝗂𝗍𝗂𝗈𝗇 TY' = TY ∧ addr' = addr
         ⟹ Transformation_Functor (DynArr addr TY) (DynArr addr' TY') T U (λ_. UNIV) (λ_. UNIV) list_all2›
       and Functional_Transformation_Functor

abbreviation ‹𝖽𝗒𝗇𝖺𝗋𝗋 ≡ 𝗌𝗍𝗋𝗎𝖼𝗍 {data: 𝗉𝗍𝗋, len: 𝖺𝗂𝗇𝗍, cap: 𝖺𝗂𝗇𝗍}›


proc len_dynarr:
  input    ‹addr ⦂ 𝗏𝖺𝗅 ℙ𝗍𝗋 𝖽𝗒𝗇𝖺𝗋𝗋❟ l ⦂ DynArr addr TY T›
  output   ‹length l ⦂ 𝗏𝖺𝗅 ℕ❟ l ⦂ DynArr addr TY T›
❴
  𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 𝗈𝗉𝖾𝗇 ；
  val ret ← addr.len ；
  ‹MAKE _ (DynArr addr _ _)› ；
  ret
❵ .


context
  fixes TY :: TY
    and T :: ‹(VAL, 'x) φ›                              ― ‹we provide a generic verification›
    and zero :: 'x
  assumes [φreason add]: ‹Semantic_Type T TY›      ― ‹specify the semantic type of T›
      and [φreason add]: ‹Semantic_Zero_Val TY T zero›  ― ‹specify the semantic zero value of T›
begin

proc get_dynarr:
  input    ‹addr ⦂ 𝗏𝖺𝗅 ℙ𝗍𝗋 𝖽𝗒𝗇𝖺𝗋𝗋❟ i ⦂ 𝗏𝖺𝗅 ℕ❟ l ⦂ DynArr addr TY T›
  premises ‹i < length l›
  output   ‹l!i ⦂ 𝗏𝖺𝗅 T❟ l ⦂ DynArr addr TY T›
❴
  𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 𝗈𝗉𝖾𝗇 ；
  addr.data[i]
  𝗆𝖺𝗄𝖾𝗌 ‹DynArr addr _ _›
❵ .


proc set_dynarr:
  input    ‹addr ⦂ 𝗏𝖺𝗅 ℙ𝗍𝗋 𝖽𝗒𝗇𝖺𝗋𝗋❟ i ⦂ 𝗏𝖺𝗅 ℕ❟ v ⦂ 𝗏𝖺𝗅 T❟ l ⦂ DynArr addr TY T›
  premises ‹i < length l›
  output   ‹l[i := v] ⦂ DynArr addr TY T›
❴
  𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 𝗈𝗉𝖾𝗇 ；
  addr.data[i] := v ； 
  𝗆𝖺𝗄𝖾𝗌 ‹l[i := v] ⦂ (DynArr addr _ _)›
❵ .

proc Max:
  input  ‹x ⦂ 𝗏𝖺𝗅 ℕ❟ y ⦂ 𝗏𝖺𝗅 ℕ›
  output ‹max x y ⦂ 𝗏𝖺𝗅 ℕ›
❴
  if (x < y) ❴ y ❵ ❴ x ❵
❵ .


proc push_dynarr:
  input    ‹addr ⦂ 𝗏𝖺𝗅 ℙ𝗍𝗋 𝖽𝗒𝗇𝖺𝗋𝗋❟ v ⦂ 𝗏𝖺𝗅 T❟ l ⦂ DynArr addr TY T›
  output   ‹l @ [v] ⦂ DynArr addr TY T›
❴
  𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 𝗈𝗉𝖾𝗇 ；
  val len ← addr.len ；
  val cap ← addr.cap ；
  if (cap = len) ❴
      val cap' ← Max(cap * 2, 1) ；
      val data' ← calloc_aN (cap') ‹T› ；
      memcpy_a (data', addr.data, len) ；
      mfree (addr.data) ；
      addr.data := data' ；
      addr.len := addr.len + 1 ；
      addr.cap := cap' ；
      data'[len] := v ；
      𝗆𝖺𝗄𝖾𝗌 ‹l@[v] ⦂ DynArr addr _ _›
  ❵ ❴
      addr.data[len] := v ；
      addr.len := len + 1 ；
      𝗆𝖺𝗄𝖾𝗌 ‹l@[v] ⦂ DynArr addr _ _›
  ❵
❵ .

proc concat_dynarr:
  input   ‹addr1 ⦂ 𝗏𝖺𝗅 ℙ𝗍𝗋 𝖽𝗒𝗇𝖺𝗋𝗋❟ addr2 ⦂ 𝗏𝖺𝗅 ℙ𝗍𝗋 𝖽𝗒𝗇𝖺𝗋𝗋❟
           l1 ⦂ DynArr addr1 TY T❟ l2 ⦂ DynArr addr2 TY T›
  output  ‹l1 @ l2 ⦂ DynArr addr1 TY T❟ l2 ⦂ DynArr addr2 TY T›
❴
  val len ← len_dynarr (addr2) ；
  iterate_a (‹0 ⦂ ℕ›, len)
            ‹λi. l1 @ take i l2 ⦂ DynArr addr1 TY T›
  ❴ → val i ；
    push_dynarr (addr1, get_dynarr (addr2, i))
  ❵
❵ .


proc pop_dynarr:
  input    ‹addr ⦂ 𝗏𝖺𝗅 ℙ𝗍𝗋 𝖽𝗒𝗇𝖺𝗋𝗋❟ v ⦂ 𝗏𝖺𝗅 T❟ l ⦂ DynArr addr TY T›
  premises ‹l ≠ []›
  output   ‹last l ⦂ 𝗏𝖺𝗅 T❟ butlast l ⦂ DynArr addr TY T›
❴
  𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 𝗈𝗉𝖾𝗇 ；
  val len ← addr.len - 1 ；
  val half_cap ← addr.cap / 2 ；
  val ret ← addr.data[len] ；
  addr.len := len ；
  if (len ≤ half_cap) ❴
    val data' ← calloc_aN (half_cap) ‹T› ；
    memcpy_a (data', addr.data, len) ；
    mfree (addr.data) ；
    addr.data := data' ；
    addr.cap := half_cap ；
    𝗆𝖺𝗄𝖾𝗌 ‹DynArr addr _ _›
  ❵
  ❴ 𝗆𝖺𝗄𝖾𝗌 ‹DynArr addr _ _› ❵
  ret
❵ .


proc new_dynarr:
  input  ‹Void›
  output ‹addr ⦂ 𝗏𝖺𝗅 ℙ𝗍𝗋 𝖽𝗒𝗇𝖺𝗋𝗋❟ [] ⦂ DynArr addr TY T 𝗌𝗎𝖻𝗃 addr. ⊤›
❴
  val ret ← calloc1 ‹⦃ data: ℙ𝗍𝗋 𝖺𝗋𝗋𝖺𝗒[0] TY, len: ℕ, cap: ℕ ⦄› ；
  ret.data := (calloc_aN (0) ‹T›) ；
  𝗆𝖺𝗄𝖾𝗌 ‹DynArr addr _ _› ；
  ret
❵ .


proc del_dynarr:
  input  ‹addr ⦂ 𝗏𝖺𝗅 ℙ𝗍𝗋 𝖽𝗒𝗇𝖺𝗋𝗋❟ l ⦂ DynArr addr TY T›
  output ‹Void›
❴
  𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 𝗈𝗉𝖾𝗇 ；
  mfree (addr.data) ；
  mfree (addr)
❵ .

proc map_dynarr:
  input  ‹addr ⦂ 𝗏𝖺𝗅 ℙ𝗍𝗋 𝖽𝗒𝗇𝖺𝗋𝗋❟ l ⦂ DynArr addr TY T›
  requires C: ‹⋀x u. 𝗉𝗋𝗈𝖼 C u ⦃ x ⦂ 𝗏𝖺𝗅[u] T ⟼ f x ⦂ 𝗏𝖺𝗅 T ⦄ ›
  output ‹map f l ⦂ DynArr addr TY T›
❴
  note [φsledgehammer_simps] = list_eq_iff_nth_eq nth_append ；

  iterate_a (‹0 ⦂ ℕ›, len_dynarr ($addr)) ‹λi. (map f (take i l) @ drop i l) ⦂ DynArr addr TY T›
  ❴ → val i ；
     set_dynarr (addr, i, C (get_dynarr (addr, i)))
  ❵
❵ .

proc exists_dynarr:
  input  ‹addr ⦂ 𝗏𝖺𝗅 ℙ𝗍𝗋 𝖽𝗒𝗇𝖺𝗋𝗋❟ l ⦂ DynArr addr TY T›
  requires C: ‹⋀x u. 𝗉𝗋𝗈𝖼 C u ⦃ x ⦂ 𝗏𝖺𝗅[u] T ⟼ P x ⦂ 𝗏𝖺𝗅 𝔹 ⦄ ›
  output ‹list_ex P l ⦂ 𝗏𝖺𝗅 𝔹❟ l ⦂ DynArr addr TY T›
❴
  var zz ← False ;;
  iterate_a (‹0 ⦂ ℕ›, len_dynarr (addr))
            ‹λi. l ⦂ DynArr addr TY T❟ list_ex P (take i l) ⦂ 𝗏𝖺𝗋[zz] 𝔹› ；
    ❴ → val i ；
      zz ∨ C (get_dynarr (addr, i)) → zz
    ❵ ；
  zz
❵ .


proc fold_map_dynarr:
  input  ‹addr ⦂ 𝗏𝖺𝗅 ℙ𝗍𝗋 𝖽𝗒𝗇𝖺𝗋𝗋❟ z0 ⦂ 𝗏𝖺𝗅 U❟ l ⦂ DynArr addr TY T›
  requires [φreason]: ‹Semantic_Type U TY⇩U›
       and C: ‹⋀x z u v. 𝗉𝗋𝗈𝖼 C u v ⦃ x ⦂ 𝗏𝖺𝗅[u] T❟ z ⦂ 𝗏𝖺𝗅[v] U ⟼ f x ⦂ 𝗏𝖺𝗅 T❟ g x z ⦂ 𝗏𝖺𝗅 U ⦄ ›
  output ‹fold g l z0 ⦂ 𝗏𝖺𝗅 U❟ map f l ⦂ DynArr addr TY T›
❴
  var zz ← z0 ；
  iterate_a (‹0 ⦂ ℕ›, len_dynarr (addr))
             ‹λi. fold g (take i l) z0 ⦂ 𝗏𝖺𝗋[zz] U❟ (map f (take i l) @ drop i l) ⦂ DynArr addr TY T›
  ❴ → val i ；
    C (get_dynarr (addr, i), zz) → val x', var zz ；
    set_dynarr (addr, i, x')
  ❵ certified by (auto simp add: list_eq_iff_nth_eq nth_append, auto_sledgehammer,
                  insert ‹i < length l›, induct i, auto simp add: take_Suc_conv_app_nth) ；
  $zz
❵ .

text ‹The Conclusions of above Certification is the following Specification Theorems›
