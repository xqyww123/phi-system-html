Theory Quicksort

theory Quicksort
  imports Phi_Semantics.PhiSem_C
          "HOL-Combinatorics.List_Permutation"
          PhiStd.PhiStd_Loop
begin

declare [[auto_sledgehammer_params = "try0 = false"]]



  proc qsort:
    input  ‹𝗏𝖺𝗅 i ⦂ 𝗌𝗅𝗂𝖼𝖾ℙ𝗍𝗋[addr:LEN] 𝗂𝗇𝗍❟
            𝗏𝖺𝗅 len ⦂ ℕ(𝗂𝗇𝗍)❟
            l ⦂ 𝗆𝖾𝗆[addr] 𝗌𝗅𝗂𝖼𝖾[i,len] ℕ(𝗂𝗇𝗍)›
    premises ‹i + len ≤ LEN›
    output ‹l' ⦂ 𝗆𝖾𝗆[addr] 𝗌𝗅𝗂𝖼𝖾[i,len] ℕ(𝗂𝗇𝗍)
            𝗌𝗎𝖻𝗃 l'. l <~~> l' ∧ sorted l'›
    is [recursive]
    is [routine]
  ❴
    if (len ≤ 1) ❴
      return
    ❵ ❴
      val pivot ← *(i + (len - 1)) ；
      var d ← 0 ；
      iterate (0,len) ‹λn. d  ⦂ 𝗏𝖺𝗋[d] ℕ(𝗂𝗇𝗍)❟
                           l' ⦂ 𝗆𝖾𝗆[addr] 𝗌𝗅𝗂𝖼𝖾[i,len] ℕ(𝗂𝗇𝗍)
                           𝗌𝗎𝖻𝗃 l' d.
                             d ≤ n ∧ l <~~> l' ∧
                             (∀k<d. l' ! k ≤ ?pivot) ∧
                             (∀k<n-d. ?pivot < l' ! (d + k)) › 
      ❴
        for n → val n ；
        *(i + n) → val x ；
        if (x ≤ pivot) ❴
          (i + n) := *(i + d) ；
          (i + d) := x ；
          d ← d + 1
        ❵ ❴
          (*comment: else, do nothing*)
        ❵ ；
      ❵
      qsort (i, d) ；
      qsort (i + d, len - d) ；
          
      holds_fact t1: ‹(∀x∈set (drop d l'). l ! (len - 1) < x)›
             and t2: ‹(∀x∈set (take d l'). x ≤ l ! (len - 1))›
             and t3[simp]: ‹set l'b = set (drop d l')›
             and t4[simp]: ‹set l'a = set (take d l')› ；
  
      return
    ❵
  ❵ .


text ‹The Conclusions of above Certification is the following Specification Theorems›

thm qsort_φapp

text ‹Semantic Representations of the Programs: ›
  
thm qsort_def

end
