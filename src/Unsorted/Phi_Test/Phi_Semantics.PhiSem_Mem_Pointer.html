lemma infinite_UNIV_int [simp]: "infinite (UNIV::int set)"
proof
  assume "finite (UNIV::int set)"
  moreover have "inj (λi::int. 2 * i)"
    by (rule injI) simp
  ultimately have "surj (λi::int. 2 * i)"
    by (rule finite_UNIV_inj_surj)
  thm finite_UNIV_inj_surj
  then obtain i :: int where "1 = 2 * i" by (rule surjE)
  then show False by presburger
qed

lemma Mem_freshness:
  ‹finite (dom f) ⟹ ∃k. f k = None ∧ memblk.layout k = TY ∧ k ≠ Null›
  unfolding dom_def
proof (clarsimp)
  assume a1: "finite {a. ∃y. f a = Some y}"
  obtain bb :: "(memblk ⇒ nat) ⇒ (memblk ⇒ bool) ⇒ nat ⇒ bool" where
    f2: "∀X28 X31 X30. bb X30 X31 X28 = (∃X32. X28 = X30 X32 ∧ X31 X32)"
    by moura
  then have f3: "∀f p. Collect (bb f p) = f ` Collect p"
    by auto
  obtain nn :: "memblk ⇒ nat" where
    f4: "∀t n. nn (MemBlk n t) = n"
    by (metis (no_types) memblk.distinct(1) memblk.exhaust memblk.inject)
  obtain bba :: "memblk ⇒ bool" where
    f5: "∀X39. bba X39 = (∃X40. f X39 = Some X40)"
    by moura
  then have "finite (Collect bba)"
    using a1 by presburger
  then show ?thesis
    using f5 f4 f3 f2 by (metis (no_types) UNIV_eq_I finite_imageI infinite_UNIV_nat mem_Collect_eq memblk.distinct(1) memblk.layout(2) option.exhaust)
qed

lemma index_offset_inj:
  assumes prems:
    ‹valid_index T index1›
    ‹valid_index T index2›
    ‹index_type index2 T = index_type index1 T›
    ‹¬ phantom_mem_semantic_type (index_type index1 T)›
  shows ‹index_offset T index1 = index_offset T index2 ⟶ index1 = index2›
proof -
  consider (either_nil) ‹index1 = [] ∨ index2 = []›
      |   (both_notnil) ‹index1 ≠ [] ∧ index2 ≠ []›
      by blast
  then show ?thesis
  proof cases
    case either_nil
      then show ?thesis using prems index_type_idem by metis
  next
    case both_notnil
    have ‹index1 ≠ index2 ⟶ index_offset T index1 ≠ index_offset T index2› (is ‹?neq ⟶ ?goal›)
    proof
      assume neq: ?neq

      have t1: ‹⋀idx1 idx2. idx1 ≠ [] ∧ idx2 ≠ [] ∧ idx1 ≠ idx2 ∧ ¬(∃d. idx1@d = idx2) ∧ ¬(∃d. idx2@d = idx1)
          ⟹ ∃i. i < length idx1 ∧ i < length idx2 ∧ idx1 ! i ≠ idx2 ! i ›
        unfolding list_eq_iff_nth_eq nth_append apply simp apply clarify
        subgoal premises prems for idx1 idx2
          apply (cases ‹length idx1 ≤ length idx2›)
          using prems(4)[THEN spec[where x=‹drop (length idx1) idx2›], simplified]
           apply (metis le_add_diff_inverse linordered_semidom_class.add_diff_inverse nth_drop)
          using prems(5)[THEN spec[where x=‹drop (length idx2) idx1›], simplified]
           by (metis add_diff_inverse_nat le_add_diff_inverse nat_le_linear nth_drop)
         done

       moreover have t2: ‹⋀i idx1 idx2. i < length idx1 ∧ i < length idx2 ∧ idx1 ! i ≠ idx2 ! i
          ⟹ ∃i. i < length idx1 ∧ i < length idx2 ∧ idx1 ! i ≠ idx2 ! i ∧ take i idx1 = take i idx2›
         unfolding list_eq_iff_nth_eq apply simp
         subgoal for i idx1 idx2
           apply (induct i arbitrary: idx1 idx2 rule: nat_less_induct)
           by (smt (verit, ccfv_threshold) min.absorb4 min_less_iff_conj)
         done

       ultimately have t3:
          ‹⋀idx1 idx2. idx1 ≠ [] ∧ idx2 ≠ [] ∧ idx1 ≠ idx2 ∧ ¬(∃d. idx1@d = idx2) ∧ ¬(∃d. idx2@d = idx1)
          ⟹ ∃i. i < length idx1 ∧ i < length idx2 ∧ idx1 ! i ≠ idx2 ! i ∧ take i idx1 = take i idx2›
         by (smt (verit, ccfv_threshold))

       note t4 = t3[of ‹index1› ‹index2›, simplified both_notnil, simplified]
         

       have ‹¬((∃d. index1@d = index2) ∨ (∃d. index2@d = index1))›
       proof
         assume A: ‹(∃d. index1 @ d = index2) ∨ (∃d. index2 @ d = index1)›
         then consider (L) ‹∃d. index1 @ d = index2› |
                       (R) ‹(∃d. index2 @ d = index1)› by blast
         then show False
         proof cases
           case L
           then obtain d where D: ‹index2 = index1@d› by blast
           then have ‹valid_index (index_type index1 T) d› using valid_index_cat prems by blast
           then have ‹d = []› using fold_append prems D by (simp add: index_type_idem)
           then show False using neq D by blast
         next
           case R
           then obtain d where D: ‹index1 = index2@d› by blast
           then have ‹valid_index (index_type index2 T) d› using valid_index_cat prems by blast
           then have ‹d = []› using fold_append prems D by (metis comp_def index_type_idem) 
           then show False using neq D by blast
         qed
       qed
      then have ‹∃base i1 idx1 i2 idx2. index1 = (base@[i1])@idx1 ∧ index2 = (base@[i2])@idx2 ∧ i1 ≠ i2›
        using neq both_notnil t4
        by (metis append.assoc append_Cons append_Nil id_take_nth_drop)
      then obtain base i1 idx1 i2 idx2 where
        obt: ‹(base@[i1])@idx1 = index1 ∧ (base@[i2])@idx2 = index2 ∧ i1 ≠ i2›
        by blast

      have valid_idx_step_i1: ‹valid_idx_step (index_type base T) i1›
        using prems valid_index_cat valid_index_cons obt by blast
      have valid_idx_step_i2: ‹valid_idx_step (index_type base T) i2›
        using prems valid_index_cat valid_index_cons obt by blast
      have ‹index_offset T index1 ≠ index_offset T index2›
        using index_offset_bound_strict[where base = ‹base@[i1]› and idx = idx1 and T = T,
                                        simplified index_offset_tail obt prems fold_tail]
              index_offset_bound_strict[where base = ‹base@[i2]› and idx = idx2 and T = T,
                                        simplified index_offset_tail obt prems fold_tail]
              idx_step_offset_no_overlap[where T = ‹index_type base T› and i = i1 and j = i2]
        by (smt (verit, ccfv_SIG) comp_def group_cancel.add1 idx_step_offset_no_overlap nat_add_left_cancel_less nat_le_linear obt order.strict_trans1 valid_idx_step_i1 valid_idx_step_i2)
      then show ?goal
        using obt prems by presburger
    qed
    then show ?thesis using prems by blast
    qed
  qed

lemma address_to_raw_inj:
    ‹valid_address addr1 ⟹
     valid_address addr2 ⟹
     address_type addr1 = address_type addr2 ⟹
     ¬ phantom_mem_semantic_type (address_type addr1) ⟹
     address_to_raw addr1 = address_to_raw addr2 ⟶ addr1 = addr2›
  unfolding address_to_raw_def valid_address_def
  by (cases addr1; cases addr2; simp; case_tac x1; case_tac x1a; simp add: phantom_mem_semantic_type_def;
      metis Valid_MemBlk_def add.commute add_leD2 address_type_def index_offset_inj index_offset_upper_bound_0 memaddr.sel(1) memaddr.sel(2) memblk.case(2) memblk.layout(2) nat_less_le order.strict_trans1 phantom_mem_semantic_type_def unat_to_size_t)
